cmake_minimum_required(VERSION 3.31.4)

project(LightYears
	VERSION 1.0.0
	DESCRIPTION "A C++ learning project via building game app"
	LANGUAGES C CXX
)

set(CMAKE_CXX_STANDARD          14)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS       OFF)

# Allow configuring subprojects that declare older cmake_minimum_required
set(CMAKE_POLICY_VERSION_MINIMUM 3.5 CACHE STRING "Allow configuring subprojects requiring older CMake")

include(FetchContent)
set(SFML_LIB_NAME SFML)

FetchContent_Declare(
	${SFML_LIB_NAME}
	GIT_REPOSITORY https://github.com/SFML/SFML
	GIT_TAG        2.6.0
	GIT_SHALLOW   FALSE
)

FetchContent_MakeAvailable(${SFML_LIB_NAME})

# creates a variable.
set(LIGHT_YEARS_ENGINE_TARGET_NAME LightYearsEngine)
set(LIGHT_YEARS_GAME_TARGET_NAME LightYearsGame)

add_subdirectory(LightYearsEngine)
add_subdirectory(LightYearsGame)
